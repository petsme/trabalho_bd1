DROP DATABASE IF EXISTS BDBSITRAB01;
CREATE DATABASE IF NOT EXISTS BDBSITRAB01;
USE BDBSITRAB01;

CREATE TABLE Usuario (
    ID INT PRIMARY KEY,
    Nome_Sobrenome VARCHAR(80),
    Sexo CHAR,
    Data_Nascimento DATE,
    RG INT,
    CPF INT,
    Senha VARCHAR(20),
    Nome_Usuario VARCHAR(80)
);

CREATE TABLE Animal (
    ID_Animal INT PRIMARY KEY,
    ID_Protector INT,
    Nome_Animal VARCHAR(80),
    Sexo_Animal CHAR,
    Data_Cadastro DATE
);

CREATE TABLE Documentacao (
    ID_animal INT,
    Descricao VARCHAR(80),
    Data_Nascimento DATE,
    Vacinacao VARCHAR(80)
);

CREATE TABLE Protector (
    ID_Protector INT PRIMARY KEY
);

CREATE TABLE Adopter (
    ID_Adopter INT PRIMARY KEY
);

CREATE TABLE Classifica (
    ID_Usuario_1 INT,
    ID_Usuario_2 INT,
    Nota INT
);

CREATE TABLE Adota (
    ID_Animal INT,
    ID_Protector INT,
    ID_Adopter INT,
    Data_Adocao DATE
);

CREATE TABLE Agenda_Visita (
    ID_Protector INT,
    ID_Adopter INT,
    ID_Animal INT,
    Data_Agendamento DATE
);

CREATE TABLE Dados_Contato (
    ID_Usuario INT,
    Num_Telefone INT,
    Endereco VARCHAR(80),
    Email VARCHAR(100)
);
